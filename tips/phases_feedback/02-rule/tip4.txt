fn configured_or(values: &[String], defaults: &[&str]) -> Vec<String> {
    let configured = values
        .iter()
        .map(|value| value.trim())
        .filter(|value| !value.is_empty())
        .map(ToOwned::to_owned)
        .collect::<Vec<_>>();

    if configured.is_empty() {
        defaults.iter().map(|value| value.to_string()).collect()
    } else {
        dedupe_non_empty(configured)
    }
}

fn dedupe_non_empty(values: Vec<String>) -> Vec<String> {
    let mut seen = std::collections::BTreeSet::new();
    let mut out = Vec::new();

    for value in values {
        let trimmed = value.trim().trim_end_matches('/').to_string();
        if trimmed.is_empty() {
            continue;
        }
        if seen.insert(trimmed.clone()) {
            out.push(trimmed);
        }
    }

    out
}

pub fn parse(text: &str) -> Result<BoundaryManifest> {

}

pub fn load(path: &Path) -> Result<BoundaryManifest> {
    parse(&fs::read_to_string(path)?)
}

pub fn parse_rust_imports(file_path: &str, text: &str, graph: &mut DependencyGraph) {
    static RUST_USE: Lazy<Regex> =
        Lazy::new(|| Regex::new(r"^\s*(?:pub\s+)?use\s+([^;]+)").unwrap());

    for (i, line) in text.lines().enumerate() {
        let Some(caps) = RUST_USE.captures(line) else {
            continue;
        };
        let Some(target) = caps.get(1) else {
            continue;
        };
        graph.add_edge(ImportEdge {
            source_file: file_path.to_string(),
            target_module: target.as_str().trim().to_string(),
            line_number: i + 1,
        });
    }
}

pub fn parse_typescript_imports(file_path: &str, text: &str, graph: &mut DependencyGraph) {
    static TS_IMPORT: Lazy<Regex> = Lazy::new(|| {
        Regex::new(r#"(?m)^\s*import\s+(?:type\s+)?(?:.+?\s+from\s+)?['"]([^'"]+)['"]"#)
            .unwrap()
    });
    static TS_REQUIRE: Lazy<Regex> =
        Lazy::new(|| Regex::new(r#"(?m)require\(\s*['"]([^'"]+)['"]\s*\)"#).unwrap());

    let mut seen = BTreeSet::new();

    for caps in TS_IMPORT.captures_iter(text) {
        let Some(module) = caps.get(1) else {
            continue;
        };
        let line = line_number_for_offset(text, module.start());
        if seen.insert((module.as_str().to_string(), line)) {
            graph.add_edge(ImportEdge {
                source_file: file_path.to_string(),
                target_module: module.as_str().to_string(),
                line_number: line,
            });
        }
    }

    for caps in TS_REQUIRE.captures_iter(text) {
        let Some(module) = caps.get(1) else {
            continue;
        };
        let line = line_number_for_offset(text, module.start());
        if seen.insert((module.as_str().to_string(), line)) {
            graph.add_edge(ImportEdge {
                source_file: file_path.to_string(),
                target_module: module.as_str().to_string(),
                line_number: line,
            });
        }
    }
}

pub fn run_semantic_oracle(ctx: &AuditContext) -> Vec<SemanticFinding> {
    let manifest = boundary_manifest(ctx).unwrap_or_default();
    let files = semantic_files(ctx);

    let mut graph = DependencyGraph::default();
    for file in &files {
        match file.suffix.as_str() {
            ".rs" => parse_rust_imports(&file.rel_path, &file.text, &mut graph),
            ".ts" | ".tsx" | ".js" | ".jsx" | ".mjs" | ".cjs" => {
                parse_typescript_imports(&file.rel_path, &file.text, &mut graph)
            }
            ".py" => parse_python_imports(&file.rel_path, &file.text, &mut graph),
            _ => {}
        }
    }

    let mut findings = Vec::new();
    findings.extend(rust_domain_boundary_findings(&files, &graph, &manifest));
    findings.extend(typescript_web_boundary_findings(ctx, &files, &graph, &manifest));
    findings.extend(python_truth_findings(&files, &graph, &manifest));
    findings.extend(direct_db_boundary_findings(&files, &manifest));
    findings.extend(streaming_boundary_findings(&files, &manifest));
    findings.extend(streaming_exception_policy_findings(&manifest));
    findings.extend(generated_zone_oracle_findings(ctx));
    findings.extend(contract_source_oracle_findings(ctx, &manifest));

    dedupe_semantic_findings(findings)
}

pub fn has_rule_hit(ctx: &AuditContext, rule_id: &str) -> bool {

        .iter()
        .any(|finding| finding.rule_id == rule_id)
}

fn rust_domain_boundary_findings(
    files: &[FileInfo],
    graph: &DependencyGraph,
    manifest: &BoundaryManifest,
) -> Vec<SemanticFinding> {
    let domain_paths = manifest.rust_domain_paths();
    let forbidden_imports = manifest.rust_forbidden_domain_imports();
    let forbidden_functions = manifest.rust_forbidden_domain_functions();

    let mut findings = Vec::new();

    for edge in &graph.edges {
        if !path_matches_any(&edge.source_file, &domain_paths) {
            continue;
        }

        let Some(marker) = forbidden_imports
            .iter()
            .find(|marker| module_starts_with_marker(&edge.target_module, marker))
        else {
            continue;
        };

        findings.push(SemanticFinding::new(
            "data",
            "HLT-006-DIRECT-DB-WRONG-LAYER",
            edge.source_file.clone(),
            Some(edge.line_number),
            format!(
                "Rust domain boundary imports forbidden side-effect module `{}`",
                edge.target_module
            ),
            vec![
                format!("use {}", edge.target_module),
                format!("matched boundary marker `{marker}`"),
            ],
            "move IO, DB, HTTP, env, clock, random, and logging work into application/adapters; keep domain code pure and inject explicit ports or value inputs",
        ));
    }

    for file in files
        .iter()
        .filter(|file| file.suffix == ".rs" && path_matches_any(&file.rel_path, &domain_paths))
    {
        for marker in &forbidden_functions {
            let Some(line) = first_line_containing(&file.text, marker) else {
                continue;
            };

            findings.push(SemanticFinding::new(
                "data",
                "HLT-006-DIRECT-DB-WRONG-LAYER",
                file.rel_path.clone(),
                Some(line),
                format!("Rust domain boundary uses forbidden side-effect marker `{marker}`"),
                vec![format!("{}:{line} contains `{marker}`", file.rel_path)],
                "extract the side-effect call into an adapter/application service and pass pure data into the domain invariant",
            ));
        }
    }

    findings
}

fn typescript_web_boundary_findings(

    files: &[FileInfo],

    manifest: &BoundaryManifest,
) -> Vec<SemanticFinding> {

    let backend_markers = manifest.typescript_backend_import_markers();

                "HLT-006-DIRECT-DB-WRONG-LAYER",

                    edge.target_module
                ),
                vec![
                    format!("import from `{}`", edge.target_module),
                    format!("matched boundary marker `{marker}`"),
                ],
                "replace direct storage/backend imports with generated API clients or a typed HTTP/RPC boundary",
            ));
        }

        if let Some(marker) = backend_markers
            .iter()
            .find(|marker| edge.target_module.to_ascii_lowercase().contains(&marker.to_ascii_lowercase()))
        {
            findings.push(SemanticFinding::new(
                "data",
                "HLT-006-DIRECT-DB-WRONG-LAYER",
                edge.source_file.clone(),
                Some(edge.line_number),
                format!(
                    "TypeScript web layer directly imports backend module `{}`",
                    edge.target_module
                ),
                vec![
                    format!("import from `{}`", edge.target_module),
                    format!("matched backend marker `{marker}`"),
                ],
                "move durable behavior behind the API/application boundary and consume only generated contracts from web code",
            ));
        }

        if contract_signal && looks_like_handwritten_contract_import(&edge.target_module) {
            findings.push(SemanticFinding::new(
                "boundary",
                "HLT-007-HANDWRITTEN-CONTRACT",
                edge.source_file.clone(),
                Some(edge.line_number),
                format!(
                    "TypeScript web layer imports likely handwritten contract surface `{}`",
                    edge.target_module

                vec![format!("import from `{}`", edge.target_module)],
                "regenerate client/types from the source contract and import from the declared generated contract zone instead of local DTO/type mirrors",
            ));
        }

        Regex::new(r"(?m)^\s*export\s+interface\s+\w*(Request|Response|Dto|DTO)\b").unwrap()
    });

    if contract_signal {
        for file in files
            .iter()
            .filter(|file| path_matches_any(&file.rel_path, &web_paths))
            .filter(|file| matches!(file.suffix.as_str(), ".ts" | ".tsx" | ".js" | ".jsx"))

            let Some(m) = DTO_SHAPE.find(&file.text) else {
                continue;
            };

            findings.push(SemanticFinding::new(
                "boundary",
                "HLT-007-HANDWRITTEN-CONTRACT",
                file.rel_path.clone(),
                Some(line_number_for_offset(&file.text, m.start())),
                "TypeScript web layer declares API request/response DTOs while generated contracts exist",
                vec![m.as_str().trim().to_string()],
                "delete the handwritten DTO and regenerate/import the contract type from the declared generated zone",
            ));
        }
    }

    findings
}

fn python_truth_findings(
    files: &[FileInfo],
    graph: &DependencyGraph,
    manifest: &BoundaryManifest,

    let product_markers = manifest.python_product_truth_markers();

    for file in files.iter().filter(|file| file.suffix == ".py") {
        if path_matches_any(&file.rel_path, &allowed_paths) {
            continue;
        }

        for edge in graph.edges.iter().filter(|edge| edge.source_file == file.rel_path) {
            if let Some(marker) = db_markers
                .iter()
                .find(|marker| module_matches_marker(&edge.target_module, marker))

                findings.push(SemanticFinding::new(
                    "python",
                    "HLT-005-PYTHON-PRODUCT-TRUTH",

                        format!("import `{}`", edge.target_module),
                        format!("matched DB marker `{marker}`"),
                    ],
                    "move product truth and production DB access into Rust application/adapters or place this file under the bounded `python/ai-service` model/data API with a typed contract",
                ));
            }
        }

        let lower = file.text.to_ascii_lowercase();
        let db_marker = db_markers
            .iter()
            .find(|marker| lower.contains(&marker.to_ascii_lowercase()));
        let product_marker = product_markers
            .iter()
            .find(|marker| lower.contains(&marker.to_ascii_lowercase()));

        if let Some(marker) = db_marker.or(product_marker) {
            let line = first_line_containing_case_insensitive(&file.text, marker);
            findings.push(SemanticFinding::new(
                "python",
                "HLT-005-PYTHON-PRODUCT-TRUTH",
                file.rel_path.clone(),
                line,
                format!(
                    "Python outside the allowed AI/data boundary appears to own product truth marker `{marker}`"

                    "{}:{} matched `{marker}`",

                    line.unwrap_or(1)

                "box Python into model/data/eval work only, expose a typed API, and move durable product truth, authz, and production DB writes into Rust/PostgreSQL",

fn streaming_exception_policy_findings(manifest: &BoundaryManifest) -> Vec<SemanticFinding> {
    let mut findings = Vec::new();

            missing.push("runtime");
        }
        if exception.owner.trim().is_empty() {
            missing.push("owner");
        }
        if !looks_like_yyyymmdd(&exception.expires) {
            missing.push("expires(YYYY-MM-DD)");
        }
        if exception.migration_path.trim().is_empty() {
            missing.push("migration_path");
        }
        if !exception_has_brownfield_reason(exception) {

        findings.push(SemanticFinding::new(
            "boundary",
            "HLT-019-STREAMING-RUNTIME-DRIFT",
            "agent/boundaries.toml",
            None,
            format!("streaming exception `{index}` lacks required metadata"),
            vec![format!(
                "missing required streaming exception fields: {}",
                missing.join(", ")
            )],
            "make the exception temporary and repairable: set runtime, owner, YYYY-MM-DD expiry, brownfield reason/classification, and migration_path",
        ));
    }

fn generated_zone_oracle_findings(ctx: &AuditContext) -> Vec<SemanticFinding> {
    let manifest_path = ctx.root.join("agent/generated-zones.toml");
    let Ok(text) = fs::read_to_string(&manifest_path) else {
        return Vec::new();
    };
    let Ok(file) = toml::from_str::<GeneratedZonesFile>(&text) else {
        return Vec::new();
    };

    let mut findings = Vec::new();

    for (index, zone) in file.zone.iter().enumerate() {
        let path = zone.path.trim();
        let source = zone.source.trim();
        let command = zone.command.trim();

        let mut missing = Vec::new();
        if path.is_empty() {
            missing.push("path");
        }
        if source.is_empty() {
            missing.push("source");
        }
        if command.is_empty() {
            missing.push("command");
        }

        if !missing.is_empty() {
            findings.push(SemanticFinding::new(
                "generated",
                "HLT-002-GENERATED-MUTATION",
                "agent/generated-zones.toml",
                Some(1),
                format!("generated zone `{index}` lacks reproducibility metadata"),
                vec![format!("missing: {}", missing.join(", "))],
                "declare non-empty `path`, `source`, and `command` so agents edit the source contract and regenerate instead of patching generated output",
            ));
            continue;
        }

        let matching_files = ctx
            .all_files
            .iter()
            .filter(|candidate| path_matches_prefix(&candidate.rel_path, path))
            .collect::<Vec<_>>();

        for generated_file in matching_files {
            if generated_file.text.trim().is_empty() {
                continue;
            }

            let header_window = generated_file
                .text
                .lines()
                .take(20)
                .collect::<Vec<_>>()
                .join("\n")
                .to_ascii_lowercase();

            let header_missing = [
                ("Generated by:", "generated by:"),
                ("Source:", "source:"),
                ("Command:", "command:"),
                ("DO NOT EDIT", "do not edit"),
            ]
            .into_iter()
            .filter_map(|(label, marker)| {
                if header_window.contains(marker) {
                    None
                } else {
                    Some(label)
                }
            })
            .collect::<Vec<_>>();

            if !header_missing.is_empty() {
                findings.push(SemanticFinding::new(
                    "generated",
                    "HLT-002-GENERATED-MUTATION",
                    generated_file.rel_path.clone(),
                    Some(1),
                    "declared generated file lacks required generated-zone header",
                    vec![
                        format!("missing header fields: {}", header_missing.join(", ")),
                        format!("source: {source}"),
                        format!("command: {command}"),
                    ],
                    "do not edit the generated file by hand; add the standard generated header in the generator/template, then regenerate from the declared source and command",
                ));
            }

            if let Some(expected) = zone
                .checksum
                .as_deref()
                .or(zone.content_sha256.as_deref())
                .map(normalize_sha256)
            {
                let Some(actual) = sha256_file(&ctx.root, &generated_file.rel_path) else {
                    continue;
                };

                if actual != expected {
                    findings.push(SemanticFinding::new(
                        "generated",
                        "HLT-002-GENERATED-MUTATION",
                        generated_file.rel_path.clone(),
                        Some(1),
                        "declared generated file checksum does not match manifest",
                        vec![format!("expected {expected}; actual {actual}")],
                        "regenerate the file from the declared source/command and update checksum metadata only as part of the generation proof",
                    ));
                }
            }
        }
    }

    findings
}

fn contract_source_oracle_findings(
    ctx: &AuditContext,
    manifest: &BoundaryManifest,
) -> Vec<SemanticFinding> {
    if has_api_drift_checks(ctx) || has_generated_contracts(ctx) {
        return Vec::new();
    }

    let generated_paths = manifest.generated_contract_paths();
    let mut findings = Vec::new();

    for file in &ctx.all_files {
        if false_positive_path(&file.rel_path) {
            continue;
        }
        if path_matches_any(&file.rel_path, &generated_paths) {
            continue;
        }
        if !is_contract_source(file) {
            continue;
        }

        findings.push(SemanticFinding::new(
            "boundary",
            "HLT-007-HANDWRITTEN-CONTRACT",
            file.rel_path.clone(),
            Some(1),
            "contract source exists without generated output or drift proof",
            vec![format!(
                "contract source `{}` has no generated-zone/proof-lane evidence",
                file.rel_path
            )],
            "declare generated contract outputs in `agent/generated-zones.toml`, add a contract drift command, and route the changed contract through the contract proof lane",
        ));
    }

    findings
}

fn has_generated_contract_signal(ctx: &AuditContext, manifest: &BoundaryManifest) -> bool {
    if has_generated_contracts(ctx) {
        return true;
    }

    let generated_paths = manifest.generated_contract_paths();
    ctx.all_files
        .iter()
        .any(|file| path_matches_any(&file.rel_path, &generated_paths))
}

fn is_contract_source(file: &FileInfo) -> bool {
    if !file.rel_path.starts_with("contracts/") {
        return false;
    }
    if file.rel_path.contains("/generated/") {
        return false;
    }

    matches!(
        file.suffix.as_str(),
        ".proto" | ".json" | ".yaml" | ".yml" | ".graphql" | ".gql"
    ) || file.text.to_ascii_lowercase().contains("openapi")
        || file.text.to_ascii_lowercase().contains("protobuf")
}

fn looks_like_handwritten_contract_import(module: &str) -> bool {
    let lower = module.to_ascii_lowercase();
    [
        "/dto",
        "/dtos",
        "/types",
        "/api-types",
        "/api",
        "generated",
    ]
    .iter()
    .any(|marker| lower.contains(marker))
        && !lower.contains("/generated/")
        && !lower.contains("contracts/generated")
}

fn module_starts_with_marker(module: &str, marker: &str) -> bool {
    let module = module.trim();
    let marker = marker.trim();
    module == marker
        || module.starts_with(marker)
        || module.starts_with(marker.trim_end_matches("::"))
}

fn module_matches_marker(module: &str, marker: &str) -> bool {
    let module = module.trim().to_ascii_lowercase();
    let marker = marker.trim().to_ascii_lowercase();

    if module == marker {
        return true;
    }
    if module.starts_with(&format!("{marker}/")) {
        return true;
    }
    if marker.contains('/') || marker.contains('@') || marker.contains("::") {
        return module.contains(&marker);
    }

    false
}

fn path_matches_any(path: &str, patterns: &[String]) -> bool {
    patterns.iter().any(|pattern| path_matches_pattern(path, pattern))
}

fn path_matches_pattern(path: &str, pattern: &str) -> bool {
    let pattern = pattern.trim().trim_end_matches('/');
    if pattern.is_empty() {
        return false;
    }

    if let Some((head, tail)) = pattern.split_once('*') {
        let head = head.trim_end_matches('/');
        let tail = tail.trim_start_matches('/').trim_end_matches('/');
        return path.starts_with(head) && (tail.is_empty() || path.contains(tail));
    }

    path == pattern || path.starts_with(&format!("{pattern}/"))
}

fn complete_streaming_exception_for(manifest: &BoundaryManifest, marker: &str) -> bool {
    manifest.streaming_exception.iter().any(|exception| {
        let runtime = exception.runtime.trim();
        if runtime.is_empty() {
            return false;
        }

        let marker = marker.to_ascii_lowercase();
        let runtime_lower = runtime.to_ascii_lowercase();

        (marker.contains(&runtime_lower) || runtime_lower.contains(&marker))
            && !exception.owner.trim().is_empty()
            && looks_like_yyyymmdd(&exception.expires)
            && !exception.migration_path.trim().is_empty()
            && exception_has_brownfield_reason(exception)
    })
}

fn exception_has_brownfield_reason(exception: &StreamingException) -> bool {
    exception
        .classification
        .as_deref()
        .or(exception.reason.as_deref())
        .map(|value| value.eq_ignore_ascii_case("brownfield"))
        .unwrap_or(false)
}

fn looks_like_yyyymmdd(value: &str) -> bool {
    static DATE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^\d{4}-\d{2}-\d{2}$").unwrap());
    DATE.is_match(value.trim())
}

fn first_line_containing(text: &str, marker: &str) -> Option<usize> {
    text.lines()
        .position(|line| line.contains(marker))
        .map(|idx| idx + 1)
}

fn first_line_containing_case_insensitive(text: &str, marker: &str) -> Option<usize> {
    let marker = marker.to_ascii_lowercase();
    text.lines()
        .position(|line| line.to_ascii_lowercase().contains(&marker))
        .map(|idx| idx + 1)
}

fn line_number_for_offset(text: &str, offset: usize) -> usize {
    text[..offset.min(text.len())]
        .bytes()
        .filter(|byte| *byte == b'\n')
        .count()
        + 1
}

fn normalize_sha256(value: &str) -> String {
    let trimmed = value.trim();
    if trimmed.starts_with("sha256:") {
        trimmed.to_string()
    } else {
        format!("sha256:{trimmed}")
    }
}

fn sha256_file(root: &Path, rel_path: &str) -> Option<String> {
    fs::read(root.join(rel_path))
        .ok()
        .map(|bytes| format!("sha256:{:x}", Sha256::digest(bytes)))
}

fn dedupe_semantic_findings(findings: Vec<SemanticFinding>) -> Vec<SemanticFinding> {
    let mut seen = BTreeSet::new();
    let mut out = Vec::new();

    for finding in findings {
        let key = (
            finding.rule_id.to_string(),
            finding.path.clone(),
            finding.line.unwrap_or(0),
            finding.problem.clone(),
        );

        if seen.insert(key) {
            out.push(finding);
        }
    }

    out.sort_by(|a, b| {
        a.path
            .cmp(&b.path)
            .then(a.line.unwrap_or(0).cmp(&b.line.unwrap_or(0)))
            .then(a.rule_id.cmp(b.rule_id))
            .then(a.problem.cmp(&b.problem))
    });

    out
}

cat > docs/boundary-oracle.md <<'MD'
# Boundary Oracle

Status: implemented semantic-oracle slice  
Owner: tools  
Last reviewed: 2026-05-03  
Applies to: `agent/boundaries.toml`, `crates/jankurai/src/audit/analyzers/ast.rs`, `crates/jankurai/src/boundaries/`

The boundary oracle makes ownership executable. It turns path policy plus shallow language evidence into stable rule findings with owners, lanes, docs, evidence kinds, confidence, and repair instructions.

## Inputs

- `agent/boundaries.toml`
- `agent/owner-map.json`
- `agent/test-map.json`
- `agent/generated-zones.toml`
- repository inventory from the audit scanner
- stable rule metadata in `crates/jankurai/src/audit/rules.rs`

## Implemented Checks

The semantic oracle is dependency-light on purpose. It parses imports and content markers instead of adding a heavyweight compiler or tree-sitter dependency to the fast lane.

### Rust domain boundary

Rust domain paths default to:

- `crates/domain`
- `crates/*/src/domain`
- `domain`

Those paths cannot import or call filesystem, environment, network, DB, HTTP, queue, clock/random, logging, or tracing side effects directly. Findings use `HLT-006-DIRECT-DB-WRONG-LAYER` because the practical repair is the same boundary move: extract side effects into application/adapters and keep domain decisions pure.

### TypeScript web boundary

Web paths default to:

- `apps/web`
- `frontend`
- `ui`
- `packages/web`
- `packages/ui`

The web layer cannot import DB packages, backend modules, storage clients, or handwritten DTO/type mirrors when generated contracts exist. Findings use either:

- `HLT-006-DIRECT-DB-WRONG-LAYER` for direct backend/storage access
- `HLT-007-HANDWRITTEN-CONTRACT` for generated-contract drift

### Python product-truth containment

Python can own model/data/eval behavior under `python/ai-service` and tooling under `tools`. Python elsewhere cannot own durable product truth, authorization, tenant/billing/user state, or production DB writes. Findings use `HLT-005-PYTHON-PRODUCT-TRUTH`.

### DB boundary

Direct DB markers outside declared `db/`, migration, constraint, or adapter paths produce `HLT-006-DIRECT-DB-WRONG-LAYER`.

### Streaming boundary

Broker clients such as Kafka, Tansu, Iggy, Fluvio, NATS, and Redis Streams must live behind declared queue adapter paths. Escapes produce `HLT-019-STREAMING-RUNTIME-DRIFT`.

Streaming exceptions must include:

- `runtime`
- `owner`
- `expires` in `YYYY-MM-DD` form
- `classification` or `reason` set to `brownfield`
- `migration_path`

Incomplete exceptions are findings on `agent/boundaries.toml`.

### Generated-zone oracle

Every `[[zone]]` row in `agent/generated-zones.toml` must have non-empty `path`, `source`, and `command`. Declared generated text/code files must expose the standard header:

```text
Generated by:
Source:
Command:
DO NOT EDIT
````

Optional `checksum` or `content_sha256` fields are honored when present. Mismatches produce `HLT-002-GENERATED-MUTATION`.

## Repair Contract

Every semantic finding should answer:

* what failed
* why it violates the ownership boundary
* where to repair
* which lane to rerun
* which docs explain the rule

The report path is:

```text
inventory -> analyzers -> rule registry -> findings -> caps -> score -> reports
```

Validation:

```bash
cargo test -p jankurai semantic_oracle -- --nocapture
cargo test -p jankurai rule_registry_smoke -- --nocapture
cargo test -p jankurai
just fast
```

# Jankurai Audit Rules

Status: implemented
Owner: tools
Last reviewed: 2026-05-03
Applies to: `crates/jankurai/src/audit/rules.rs`

The canonical rule registry lives in Rust at `crates/jankurai/src/audit/rules.rs`. Each rule carries:

* stable ID
* name
* category
* TLR
* default severity
* default proof lane
* owner hint
* docs URL
* evidence kind
* repairability
* default confidence
* optional hard-cap impact

The registry is intentionally executable rather than prose-only. Report construction looks up emitted `rule_id` values through the registry so findings inherit lane, owner hint, docs URL, TLR, evidence kind, and confidence.

## HLT-001-DEAD-MARKER

Future-hostile markers, placeholder language, TODO/stub behavior, or dead-language tokens in runtime/product code.

Repair: implement the behavior, delete dead code, rename the real state, or add a typed exception with owner and expiry.

## HLT-002-GENERATED-MUTATION

Generated output lacks source/command metadata, generated headers, checksum consistency, or source-of-truth regeneration evidence.

Repair: edit the source contract or generator, not the generated output; regenerate and record proof.

## HLT-003-OWNERLESS-PATH

A path has no owner-map route.

Repair: add the narrowest stable prefix to `agent/owner-map.json`.

## HLT-004-UNMAPPED-PROOF

A path has no test-map proof lane or the repo lacks a deterministic fast lane.

Repair: route the path through `agent/test-map.json` and keep the command runnable.

## HLT-005-PYTHON-PRODUCT-TRUTH

Python outside the bounded AI/data service owns durable product behavior, authorization, or production DB access.

Repair: move truth into Rust/PostgreSQL and keep Python to model/data/eval APIs.

## HLT-006-DIRECT-DB-WRONG-LAYER

DB, storage, IO, or durable side-effect logic appears in a layer that must not own it.

Repair: move DB clients and SQL into declared DB/adapters paths; keep domain pure and web generated-client-only.

## HLT-007-HANDWRITTEN-CONTRACT

API/client/request/response contract mirrors are handwritten where generated contracts should exist.

Repair: regenerate contracts from source and import from declared generated zones.

## HLT-008-FALSE-GREEN-RISK

A proof lane passes without actually proving changed behavior.

Repair: replace skipped, focused, tautological, or snapshot-only tests with semantic assertions.

## HLT-009-GENERATED-SECURITY

Generated security-sensitive code lacks security proof.

Repair: run the security lane and add negative tests/threat evidence.

## HLT-010-SECRET-SPRAWL

Secret-like material appears in repository text.

Repair: remove and rotate, then add secret scanning and artifact/transcript review.

## HLT-011-PROMPT-INJECTION

Untrusted content can affect trusted agent/tool policy.

Repair: enforce source hierarchy and validate tool calls against repo policy.

## HLT-012-OVERBROAD-AGENCY

Agent/tool permissions exceed lane scope.

Repair: use least-privilege profiles and approval gates.

## HLT-013-RENDERED-UX-GAP

User-facing UI lacks rendered proof.

Repair: add Storybook/Playwright/screenshots/geometry/baseline evidence.

## HLT-014-A11Y-GAP

UI lacks accessibility proof or has accessibility violations.

Repair: fix axe/WCAG findings and emit accessibility artifacts.

## HLT-015-CONTEXT-SETUP-GAP

Setup/context routing is not deterministic.

Repair: shorten root guidance, route detail through machine-readable maps, and keep commands local.

## HLT-016-SUPPLY-CHAIN-DRIFT

Dependency, action, image, package, or provenance change lacks scan evidence.

Repair: run dependency, SBOM/provenance, and workflow hardening checks.

## HLT-017-OPAQUE-OBSERVABILITY

Failures lack agent-repairable telemetry or typed exception guidance.

Repair: add structured errors, reason codes, docs URLs, common fixes, and receipts.

## HLT-018-PERF-CONCURRENCY-DRIFT

Performance or concurrency risk lacks proof.

Repair: add benchmarks, traces, budgets, cancellation/idempotency, or targeted fast proof.

## HLT-019-STREAMING-RUNTIME-DRIFT

Broker client or streaming runtime escapes queue adapter boundaries, or an exception lacks owner/expiry/migration proof.

Repair: move broker clients behind adapters/generated event contracts or complete the brownfield exception.

## HLT-020-CI-HARDENING-GAP

CI workflow permissions, pinning, or secret handling is unsafe.

Repair: pin actions, narrow permissions, and prevent secret echo/leakage.

## HLT-021-DESTRUCTIVE-MIGRATION

Destructive SQL under migration paths lacks documented safety evidence.

Repair: add rollback/down/backfill/lock/staged-deploy evidence and run `db-migration-analyze`.
