**Phase name:** Phase 12: Benchmark Certification And Governance
**Target state:** hardened

I read this against the moonshot loop: intent → bounded agents → proof lanes → evidence → expired exceptions → reusable primitives. The important Phase 12 gap is not another local report; it is making certification publicly meaningful with machine-readable evidence, badge output, digest-bound attestations, and CI publication. That matches the moonshot’s demand for durable, auditable evidence and the success criterion that a fresh agent can prove and leave evidence without learning the whole codebase first. ([GitHub][1])

The current Phase 12 file says the existing implementation already emits schema-valid benchmark, certification, and governance artifacts, but explicitly deferred external signing, hosted dashboard, and public badge publishing. ([GitHub][2]) The Phase 00 router also names Phase 12 public evidence as one of the remaining high-leverage deferred slices. ([GitHub][3]) The patch below finishes that slice by adding a **public evidence bundle**, **badge JSON/SVG**, **local digest attestation**, and **CI artifact attestation step**. GitHub’s current artifact-attestation guidance requires workflow permissions such as `id-token: write`, `contents: read`, and `attestations: write`, then an `actions/attest@v4` step with a `subject-path`; the workflow below follows that pattern. ([GitHub Docs][4])

pub fn run(args: CertifyArgs) -> Result<()> {
    let certification = build_certification(&args.repo)?;

    if let Some(path) = args.out.as_deref() {
        validation::write_json(&args.repo, ArtifactSchema::Certification, path, &certification)?;
    } else {
        validation::validate_serializable(&args.repo, ArtifactSchema::Certification, &certification)?;
        println!("{}", serde_json::to_string_pretty(&certification)?);
    }

    if let Some(path) = args.md.as_deref() {
        crate::render::write_markdown(path, &render_markdown(&certification))?;
    }

    if args.wants_public_outputs() {
        let mut bundle = build_certification_bundle(&args.repo, certification)?;
        apply_publication_paths(&mut bundle, &args);

        if let Some(path) = args.bundle.as_deref() {
            validation::write_json(
                &args.repo,
                ArtifactSchema::CertificationBundle,
                path,
                &bundle,
            )?;
        }

        if let Some(path) = args.public_md.as_deref() {
            crate::render::write_markdown(path, &render_public_markdown(&bundle))?;
        }

        let badge = badge_endpoint(&bundle.badge);
        if let Some(path) = args.badge_json.as_deref() {
            validation::write_json(&args.repo, ArtifactSchema::BadgeEndpoint, path, &badge)?;
        }

        if let Some(path) = args.badge_svg.as_deref() {
            crate::render::write_markdown(path, &render_badge_svg(&badge))?;
        }

        if let Some(path) = args.attestation.as_deref() {
            validation::write_json(
                &args.repo,
                ArtifactSchema::CertificationAttestation,
                path,
                &bundle.attestation,
            )?;
        }
    }

    Ok(())
}

pub fn build_certification(repo: &Path) -> Result<Certification> {
    let release = load_release_data(repo)?;
    let score_path = repo.join("agent/repo-score.json");
    let score = read_repo_score(repo)?;
    let evidence_root = if score_path.exists() {
        repo.to_path_buf()
    } else {
        workspace_root()
    };

    let (score_value, caps, mut findings_summary) = match score.as_ref() {
        Some(summary) => (
            summary.score,
            summary.caps.clone(),
            FindingsSummary {
                critical: summary.findings.critical,
                high: summary.findings.high,
                medium: summary.findings.medium,
                low: summary.findings.low,
                notes: vec![
                    format!("repo score artifact: {}", score_path.display()),
                    "certification is local evidence until a CI attestation binds the public bundle"
                        .to_string(),
                ],
            },
        ),
        None => (
            0,
            Vec::new(),
            FindingsSummary {
                critical: 0,
                high: 0,
                medium: 0,
                low: 0,
                notes: vec![
                    format!("score artifact missing: {}", score_path.display()),
                    "certification is local evidence until a CI attestation binds the public bundle"
                        .to_string(),
                ],
            },
        ),
    };

    if let Some(summary) = score.as_ref() {
        if summary.caps.is_empty() {
            findings_summary
                .notes
                .push("no hard caps reported in the score artifact".to_string());
        }
    }

    let proof_receipt_index = collect_receipts(&evidence_root)?;
    let security_receipt_index = collect_security_receipts(&evidence_root)?;
    let ux_receipt_index = collect_ux_receipts(&evidence_root)?;
    let contract_db_receipt_index = collect_contract_db_receipts(&evidence_root)?;
    let exceptions = collect_exceptions(&evidence_root)?;

    let conformance_level = conformance_level(
        score_value,
        &caps,
        &findings_summary,
        proof_receipt_index.len(),
        security_receipt_index.len(),
        contract_db_receipt_index.len(),
    );

    Ok(Certification {
        schema_version: release.schema_version,
        generated_at: now_string(),
        repo: repo.display().to_string(),
        standard_version: release.standard_version,
        auditor_version: release.auditor_version,
        paper_edition: release.paper_edition,
        target_stack_id: release.target_stack_id,
        score: score_value,
        conformance_level,
        caps,
        findings_summary,
        proof_receipt_index,
        security_receipt_index,
        ux_receipt_index,
        contract_db_receipt_index,
        exceptions,
        provenance: Provenance {
            attestation: "local-digest-bundle; CI may add GitHub artifact attestation".to_string(),
            signature: None,
            build_url: github_build_url(),
        },
    })
}

pub fn build_certification_bundle(
    repo: &Path,
    certification: Certification,
) -> Result<CertificationBundle> {
    let suite = bench::build_benchmark_suite(repo)?;
    validation::validate_serializable(repo, ArtifactSchema::BenchmarkSuite, &suite)?;
    let benchmark_report = bench::build_benchmark_report(repo, &suite)?;
    validation::validate_serializable(repo, ArtifactSchema::BenchmarkReport, &benchmark_report)?;

    let governance_policy = govern::build_governance_policy(repo)?;
    validation::validate_serializable(repo, ArtifactSchema::GovernancePolicy, &governance_policy)?;

    let certification_digest =
        digest_virtual_json("virtual:certification", "certification", &certification)?;
    let benchmark_digest =
        digest_virtual_json("virtual:benchmark-report", "benchmark-report", &benchmark_report)?;
    let governance_digest = digest_virtual_json(
        "virtual:governance-policy",
        "governance-policy",
        &governance_policy,
    )?;

    let mut artifact_digests = vec![
        certification_digest.clone(),
        benchmark_digest.clone(),
        governance_digest.clone(),
    ];
    collect_public_digest_sources(repo, &certification, &mut artifact_digests)?;
    let artifact_digests = dedup_digests(artifact_digests);

    let badge = certification_badge(&certification);

    let attestation = CertificationAttestation {
        schema_version: certification.schema_version.clone(),
        generated_at: now_string(),
        subject: certification.repo.clone(),
        issuer: "jankurai certify".to_string(),
        predicate_type: "https://jankurai.dev/attestations/certification/v1".to_string(),
        statement: AttestationStatement {
            standard_version: certification.standard_version.clone(),
            auditor_version: certification.auditor_version.clone(),
            paper_edition: certification.paper_edition.clone(),
            target_stack_id: certification.target_stack_id.clone(),
            score: certification.score,
            conformance_level: certification.conformance_level.clone(),
            certification_sha256: certification_digest.sha256.clone(),
            benchmark_report_sha256: benchmark_digest.sha256.clone(),
            governance_policy_sha256: governance_digest.sha256.clone(),
            notes: vec![
                "This local attestation binds Phase 12 evidence by SHA-256 digest.".to_string(),
                "GitHub Actions can additionally sign the public bundle with actions/attest."
                    .to_string(),
            ],
        },
        artifact_digests: artifact_digests.clone(),
    };

    Ok(CertificationBundle {
        schema_version: certification.schema_version.clone(),
        generated_at: now_string(),
        repo: certification.repo.clone(),
        standard_version: certification.standard_version.clone(),
        auditor_version: certification.auditor_version.clone(),
        paper_edition: certification.paper_edition.clone(),
        target_stack_id: certification.target_stack_id.clone(),
        score: certification.score,
        conformance_level: certification.conformance_level.clone(),
        badge,
        certification,
        benchmark_report,
        governance_policy,
        artifact_digests,
        attestation,
        publication: default_publication(),
        hardening_checks: vec![
            "benchmark report validates against schemas/benchmark-report.schema.json".to_string(),
            "certification validates against schemas/certification.schema.json".to_string(),
            "governance policy validates against schemas/governance-policy.schema.json".to_string(),
            "public bundle validates against schemas/certification-bundle.schema.json".to_string(),
            "badge endpoint validates against schemas/badge-endpoint.schema.json".to_string(),
            "local attestation validates against schemas/certification-attestation.schema.json"
                .to_string(),
            "CI uploads the public evidence bundle and attests it on push".to_string(),
        ],
    })
}

fn conformance_level(
    score: i32,
    caps: &[String],
    findings: &FindingsSummary,
    proof_receipts: usize,
    security_receipts: usize,
    contract_db_receipts: usize,
) -> String {
    let clean = score >= 93 && caps.is_empty() && findings.high == 0 && findings.critical == 0;
    let release_evidence_present =
        proof_receipts > 0 && security_receipts > 0 && contract_db_receipts > 0;

    if clean && release_evidence_present {
        "HL4".to_string()
    } else if clean {
        "HL3".to_string()
    } else if score >= 85 && findings.critical == 0 {
        "HL2".to_string()
    } else if score > 0 {
        "HL1".to_string()
    } else {
        "HL0".to_string()
    }
}

fn collect_receipts(root: &Path) -> Result<Vec<String>> {
    let mut paths = Vec::new();

    for candidate in [
        "target/jankurai/evidence-index.json",
        "target/jankurai/p10-cell-registry-lane.json",
        "target/jankurai/p10-audit-log-proof.json",
        "target/jankurai/p12-benchmark-certification-lane.json",
    ] {
        if root.join(candidate).is_file() {
            paths.push(candidate.to_string());
        }
    }

    paths.extend(collect_matching_json(root, "target/jankurai", |name| {
        name.contains("proof") || name.contains("lane") || name.contains("evidence-index")
    })?);
    paths.extend(collect_matching_json(
        root,
        "target/jankurai/proof-receipts",
        |name| name.ends_with(".json"),
    )?);

    Ok(dedup(paths))
}

        "examples/perfect-web-api-db/db/constraints/001_accounts.sql",
    ] {
        if root.join(candidate).is_file() {
            paths.push(candidate.to_string());
        }
    }

    Ok(dedup(paths))
}

fn collect_exceptions(root: &Path) -> Result<Vec<String>> {
    let mut paths = Vec::new();

    if root.join("docs/exceptions").exists() {
        paths.push("docs/exceptions/".to_string());
    }

    let candidate = "examples/perfect-web-api-db/docs/exceptions.md";
    if root.join(candidate).is_file() {
        paths.push(candidate.to_string());
    }

    Ok(dedup(paths))
}

fn collect_matching_json<F>(root: &Path, rel_dir: &str, predicate: F) -> Result<Vec<String>>
where
    F: Fn(&str) -> bool,
{
    let mut paths = Vec::new();
    let dir = root.join(rel_dir);
    if !dir.exists() {
        return Ok(paths);
    }

    for entry in fs::read_dir(&dir).with_context(|| format!("read {}", dir.display()))? {
        let entry = entry?;
        let path = entry.path();

        if path.is_dir() {
            continue;
        }

        let file_name = entry.file_name();
        let file_name = file_name.to_string_lossy();

        if file_name.ends_with(".json") && predicate(&file_name) {
            paths.push(format!("{rel_dir}/{file_name}"));
        }
    }

    Ok(dedup(paths))
}

fn collect_public_digest_sources(
    repo: &Path,
    certification: &Certification,
    out: &mut Vec<ArtifactDigest>,
) -> Result<()> {
    for (path, kind) in [
        ("agent/repo-score.json", "repo-score"),
        ("agent/repo-score.md", "repo-score-markdown"),
        ("agent/standard-version.toml", "standard-version"),
        ("agent/proof-lanes.toml", "proof-lanes"),
        ("agent/test-map.json", "test-map"),
        ("agent/owner-map.json", "owner-map"),
        ("agent/generated-zones.toml", "generated-zones"),
        (".github/workflows/jankurai.yml", "ci-workflow"),
        ("schemas/certification.schema.json", "schema"),
        ("schemas/benchmark-report.schema.json", "schema"),
        ("schemas/governance-policy.schema.json", "schema"),
        ("schemas/certification-bundle.schema.json", "schema"),
        ("schemas/certification-attestation.schema.json", "schema"),
        ("schemas/badge-endpoint.schema.json", "schema"),
    ] {
        digest_existing_file(repo, path, kind, out)?;
    }

    for path in certification
        .proof_receipt_index
        .iter()
        .chain(certification.security_receipt_index.iter())
        .chain(certification.ux_receipt_index.iter())
        .chain(certification.contract_db_receipt_index.iter())
        .chain(certification.exceptions.iter())
    {
        digest_existing_file(repo, path, "evidence-receipt", out)?;
    }

    Ok(())
}

fn digest_existing_file(
    root: &Path,
    rel_path: &str,
    kind: &str,
    out: &mut Vec<ArtifactDigest>,
) -> Result<()> {
    let path = root.join(rel_path);
    if !path.is_file() {
        return Ok(());
    }

    let bytes = fs::read(&path).with_context(|| format!("read {}", path.display()))?;
    out.push(ArtifactDigest {
        path: rel_path.replace('\\', "/"),
        kind: kind.to_string(),
        sha256: digest_bytes(&bytes),
        bytes: bytes.len() as u64,
    });

    Ok(())
}

fn digest_virtual_json<T: Serialize>(path: &str, kind: &str, value: &T) -> Result<ArtifactDigest> {
    let bytes = serde_json::to_vec_pretty(value)?;
    Ok(ArtifactDigest {
        path: path.to_string(),
        kind: kind.to_string(),
        sha256: digest_bytes(&bytes),
        bytes: bytes.len() as u64,
    })
}

fn digest_bytes(bytes: &[u8]) -> String {
    format!("sha256:{:x}", Sha256::digest(bytes))
}

fn dedup(mut values: Vec<String>) -> Vec<String> {
    values.sort();
    values.dedup();
    values
}

fn dedup_digests(mut values: Vec<ArtifactDigest>) -> Vec<ArtifactDigest> {
    values.sort_by(|left, right| {
        left.path
            .cmp(&right.path)
            .then_with(|| left.kind.cmp(&right.kind))
    });
    values.dedup_by(|left, right| left.path == right.path && left.kind == right.kind);
    values
}

fn badge_color(certification: &Certification) -> String {
    if certification.findings_summary.critical > 0
        || certification.findings_summary.high > 0
        || !certification.caps.is_empty()
    {
        "red".to_string()
    } else if certification.score >= 93 {
        "brightgreen".to_string()
    } else if certification.score >= 85 {
        "green".to_string()
    } else if certification.score >= 70 {
        "yellow".to_string()
    } else {
        "orange".to_string()
    }
}

    let label = escape_xml(&badge.label);
    let message = escape_xml(&badge.message);
    let fill = badge_color_hex(&badge.color);
    let label_width = 74_i32;
    let message_width = 20_i32 + (message.chars().count() as i32 * 7);
    let width = label_width + message_width;

    format!(

  <text x="370" y="140" transform="scale(.1)" fill="#fff" textLength="640">{label}</text>

  <text x="{message_x_shadow}" y="140" transform="scale(.1)" fill="#fff">{message}</text>

        label = label,
        message = message,

        label_width = label_width,

fn badge_color_hex(color: &str) -> &'static str {
    match color {
        "brightgreen" => "#44cc11",
        "green" => "#97ca00",
        "yellow" => "#dfb317",
        "orange" => "#fe7d37",
        "red" => "#e05d44",
        _ => "#007ec6",
    }
}

fn escape_xml(input: &str) -> String {
    input
        .replace('&', "&amp;")
        .replace('<', "&lt;")
        .replace('>', "&gt;")
        .replace('"', "&quot;")
        .replace('\'', "&apos;")
}

use jankurai::validation::{self, ArtifactSchema};
use serde_json::Value;
use std::fs;
use std::path::{Path, PathBuf};
use std::process::Command;

fn repo_root() -> PathBuf {
    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("..").join("..")
}

fn binary_path() -> PathBuf {
    PathBuf::from(env!("CARGO_BIN_EXE_jankurai"))
}

fn read_json(path: &Path) -> Value {
    serde_json::from_str(&fs::read_to_string(path).expect("read json")).expect("parse json")
}

#[test]
fn certify_public_evidence_outputs_validate_and_are_digest_bound() {
    let repo = repo_root();
    let tmp = tempfile::tempdir().unwrap();

    let certification = tmp.path().join("certification.json");
    let certification_md = tmp.path().join("certification.md");
    let bundle = tmp.path().join("jankurai-certification-bundle.json");
    let public_md = tmp.path().join("jankurai-public-report.md");
    let badge_svg = tmp.path().join("jankurai-badge.svg");
    let badge_json = tmp.path().join("jankurai-badge.json");
    let attestation = tmp.path().join("jankurai-attestation.json");

    let output = Command::new(binary_path())
        .arg("certify")
        .arg(&repo)
        .arg("--out")
        .arg(&certification)
        .arg("--md")
        .arg(&certification_md)
        .arg("--bundle")
        .arg(&bundle)
        .arg("--public-md")
        .arg(&public_md)
        .arg("--badge-svg")
        .arg(&badge_svg)
        .arg("--badge-json")
        .arg(&badge_json)
        .arg("--attestation")
        .arg(&attestation)
        .output()
        .expect("spawn jankurai certify");

    assert!(
        output.status.success(),
        "certify failed\nstdout:\n{}\nstderr:\n{}",
        String::from_utf8_lossy(&output.stdout),
        String::from_utf8_lossy(&output.stderr)
    );

    let certification_json = read_json(&certification);
    let bundle_json = read_json(&bundle);
    let badge_json_value = read_json(&badge_json);
    let attestation_json = read_json(&attestation);

    validation::validate_value(&repo, ArtifactSchema::Certification, &certification_json).unwrap();
    validation::validate_value(&repo, ArtifactSchema::CertificationBundle, &bundle_json).unwrap();
    validation::validate_value(&repo, ArtifactSchema::BadgeEndpoint, &badge_json_value).unwrap();
    validation::validate_value(
        &repo,
        ArtifactSchema::CertificationAttestation,
        &attestation_json,
    )
    .unwrap();

    assert_eq!(bundle_json["certification"]["score"], certification_json["score"]);
    assert_eq!(
        bundle_json["attestation"]["statement"]["certification_sha256"],
        attestation_json["statement"]["certification_sha256"]
    );
    assert!(bundle_json["artifact_digests"].as_array().unwrap().len() >= 6);
    assert!(bundle_json["artifact_digests"]
        .as_array()
        .unwrap()
        .iter()
        .all(|digest| digest["sha256"].as_str().unwrap().starts_with("sha256:")));
    assert!(badge_json_value["message"]
        .as_str()
        .unwrap()
        .contains(bundle_json["conformance_level"].as_str().unwrap()));

    let svg = fs::read_to_string(&badge_svg).unwrap();
    assert!(svg.starts_with("<svg"));
    assert!(svg.contains("jankurai"));

    let public_report = fs::read_to_string(&public_md).unwrap();
    assert!(public_report.contains("# jankurai Public Certification Evidence"));
    assert!(public_report.contains("Artifact Digests"));
    assert!(public_report.contains("Verification Commands"));
}
