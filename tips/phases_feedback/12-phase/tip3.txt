**Phase name:** Phase 12: Benchmark Certification And Governance

I’m treating this as a Phase 12 hardening slice, not a Phase 13 mutation slice. The repo’s moonshot is the loop `intent -> bounded agents -> proof lanes -> evidence -> expired exceptions -> reusable primitives`, with durable, machine-readable evidence as a core requirement. ([GitHub][1]) Phase 12 already has local `bench`, `certify`, and `govern` artifacts, but the phase text explicitly leaves external signing, hosted dashboard, and public badge publishing deferred. ([GitHub][2]) The router also says Phase 12’s remaining leverage is “public evidence” after the local surfaces exist. ([GitHub][3]) The release plan separately calls for a compliance badge, benchmark publication, raw evidence, and governance roles/policy, so this patch finishes that residual by adding a release-grade static publication bundle without making a hosted service mandatory. ([GitHub][4])

I could not run this patch in my local execution container because it could not resolve `github.com`, but the script below is designed to be pasted from the repo root and includes the proof commands, phase log start/finish entries, schema updates, CLI code, tests, workflow update, and phase status update.

pub fn run(args: CertifyArgs) -> Result<()> {
    let certification = build_certification_with_options(
        &args.repo,
        BuildCertifyOptions {
            signature: args.signature,
            build_url: args.build_url,
            attestation: args.attestation,
        },
    )?;

    if let Some(path) = args.out.as_deref() {
        validation::write_json(&args.repo, ArtifactSchema::Certification, path, &certification)?;
    } else {
        validation::validate_serializable(&args.repo, ArtifactSchema::Certification, &certification)?;
        println!("{}", serde_json::to_string_pretty(&certification)?);
    }

    if let Some(path) = args.md.as_deref() {
        crate::render::write_markdown(path, &render_markdown(&certification))?;
    }

    Ok(())
}

pub fn build_certification(repo: &Path) -> Result<Certification> {
    build_certification_with_options(repo, BuildCertifyOptions::default())

pub fn build_certification_with_options(
    repo: &Path,
    options: BuildCertifyOptions,
) -> Result<Certification> {
    let release = load_release_data(repo)?;
    let score_path = repo.join("agent/repo-score.json");
    let score = read_repo_score(repo)?;
    let evidence_root = if score_path.exists() {
        repo.to_path_buf()
    } else {
        workspace_root()
    };

    let (score_value, caps, mut findings_summary) = match score.as_ref() {
        Some(summary) => (
            summary.score,
            summary.caps.clone(),
            FindingsSummary {
                critical: summary.findings.critical,
                high: summary.findings.high,
                medium: summary.findings.medium,
                low: summary.findings.low,
                notes: vec![
                    format!("repo score artifact: {}", score_path.display()),
                    "certification includes content-addressed local evidence".to_string(),
                ],
            },
        ),
        None => (
            0,
            Vec::new(),
            FindingsSummary {
                critical: 0,
                high: 0,
                medium: 0,
                low: 0,
                notes: vec![
                    format!("score artifact missing: {}", score_path.display()),
                    "certification includes content-addressed local evidence".to_string(),
                ],
            },
        ),
    };

    if let Some(summary) = score.as_ref() {
        if summary.caps.is_empty() {
            findings_summary
                .notes
                .push("no hard caps reported in the score artifact".to_string());
        }
    }

    let proof_receipt_index = collect_receipts(&evidence_root)?;
    let security_receipt_index = collect_security_receipts(&evidence_root)?;
    let ux_receipt_index = collect_ux_receipts(&evidence_root)?;
    let contract_db_receipt_index = collect_contract_db_receipts(&evidence_root)?;
    let exceptions = collect_exceptions(&evidence_root)?;
    let conformance_level = conformance_level(score_value, &caps, &findings_summary);

    let evidence_artifacts = collect_certification_evidence_artifacts(
        &evidence_root,
        &proof_receipt_index,
        &security_receipt_index,
        &ux_receipt_index,
        &contract_db_receipt_index,
        &exceptions,
    )?;
    let attestation_subject_digest = evidence_subject_digest(
        &evidence_artifacts,
        &release.standard_version,
        &release.auditor_version,
        &release.paper_edition,
        &release.target_stack_id,
        score_value,
        &conformance_level,
    );
    let evidence_summary = evidence_summary(&evidence_artifacts, &attestation_subject_digest);

    let signature = options
        .signature
        .or_else(|| std::env::var("JANKURAI_CERTIFICATION_SIGNATURE").ok());
    let build_url = options
        .build_url
        .or_else(|| std::env::var("JANKURAI_CERTIFICATION_BUILD_URL").ok())
        .or_else(ci_build_url);
    let attestation = options
        .attestation
        .unwrap_or_else(|| "local-content-addressed-attestation".to_string());

    Ok(Certification {
        schema_version: release.schema_version,
        generated_at: now_string(),
        repo: repo.display().to_string(),

fn push_artifact(
    root: &Path,
    seen: &mut BTreeSet<String>,
    artifacts: &mut Vec<CertificationEvidenceArtifact>,
    path: &str,
    kind: &str,
    schema: Option<&str>,
    required: bool,
) -> Result<()> {
    let key = format!("{kind}:{path}");
    if !seen.insert(key) {
        return Ok(());
    }
    artifacts.push(certification_artifact(root, path, kind, schema, required)?);
    Ok(())
}

fn certification_artifact(
    root: &Path,
    rel_path: &str,
    kind: &str,
    schema: Option<&str>,
    required: bool,
) -> Result<CertificationEvidenceArtifact> {
    let path = root.join(rel_path);
    let exists = path.exists();
    let digest = if exists {
        Some(file_digest(&path)?)
    } else {
        None
    };
    let size_bytes = if exists && path.is_file() {
        Some(
            fs::metadata(&path)
                .with_context(|| format!("metadata {}", path.display()))?
                .len(),
        )
    } else {
        None
    };
    let status = if exists {
        "present"
    } else if required {
        "missing-required"
    } else {
        "missing-optional"
    };

    Ok(CertificationEvidenceArtifact {
        path: rel_path.to_string(),
        kind: kind.to_string(),
        schema: schema.map(str::to_string),
        digest,
        size_bytes,
        status: status.to_string(),
        required,
    })
}

fn evidence_summary(
    artifacts: &[CertificationEvidenceArtifact],
    attestation_subject_digest: &str,
) -> EvidenceSummary {
    let total = artifacts.len() as i32;
    let present = artifacts
        .iter()
        .filter(|artifact| artifact.status == "present")
        .count() as i32;
    let missing = total - present;
    let required_missing = artifacts
        .iter()
        .filter(|artifact| artifact.required && artifact.status != "present")
        .count() as i32;

    EvidenceSummary {
        total_artifacts: total,
        present_artifacts: present,
        missing_artifacts: missing,
        required_missing_artifacts: required_missing,
        digest_algorithm: "sha256".to_string(),
        attestation_subject_digest: attestation_subject_digest.to_string(),
    }
}

fn evidence_subject_digest(
    artifacts: &[CertificationEvidenceArtifact],
    standard_version: &str,
    auditor_version: &str,
    paper_edition: &str,
    target_stack_id: &str,
    score: i32,
    conformance_level: &str,
) -> String {
    let mut rows: Vec<String> = artifacts
        .iter()
        .map(|artifact| {
            format!(
                "{}|{}|{}|{}|{}|{}",
                artifact.kind,
                artifact.path,
                artifact.schema.as_deref().unwrap_or("none"),
                artifact.digest.as_deref().unwrap_or("missing"),
                artifact.status,
                artifact.required
            )
        })
        .collect();
    rows.sort();

    let mut hasher = Sha256::new();
    for header in [
        format!("standard_version={standard_version}"),
        format!("auditor_version={auditor_version}"),
        format!("paper_edition={paper_edition}"),
        format!("target_stack_id={target_stack_id}"),
        format!("score={score}"),
        format!("conformance_level={conformance_level}"),
    ] {
        hasher.update(header.as_bytes());
        hasher.update(b"\n");
    }

    for row in rows {
        hasher.update(row.as_bytes());
        hasher.update(b"\n");
    }

    format!("sha256:{:x}", hasher.finalize())
}

fn collect_matching_json<F>(root: &Path, rel_dir: &str, predicate: F) -> Result<Vec<String>>
where
    F: Fn(&str) -> bool,
{
    let mut paths = Vec::new();
    let dir = root.join(rel_dir);
    if !dir.exists() {
        return Ok(paths);
    }

    for entry in fs::read_dir(&dir)? {
        let entry = entry?;
        let file_name = entry.file_name();
        let file_name = file_name.to_string_lossy();
        if predicate(&file_name) {
            paths.push(format!("{rel_dir}/{file_name}"));
        }
    }

    Ok(dedup(paths))
}

fn dedup(mut values: Vec<String>) -> Vec<String> {
    values.sort();
    values.dedup();
    values
}

fn file_digest(path: &Path) -> Result<String> {
    let bytes = fs::read(path).with_context(|| format!("read {}", path.display()))?;
    Ok(format!("sha256:{:x}", Sha256::digest(bytes)))
}

fn ci_build_url() -> Option<String> {
    let server = std::env::var("GITHUB_SERVER_URL").ok()?;
    let repo = std::env::var("GITHUB_REPOSITORY").ok()?;
    let run = std::env::var("GITHUB_RUN_ID").ok()?;
    Some(format!("{server}/{repo}/actions/runs/{run}"))
}

fn now_string() -> String {
    SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs()
        .to_string()

    let _ = writeln!(out);

            artifact.kind,
            artifact.path,
            artifact.sha256,
            artifact.summary.as_deref().unwrap_or("")

fn render_sha256sums(public_dir: &Path, rel_paths: &[&str]) -> Result<String> {
    let mut out = String::new();

    for rel_path in rel_paths {
        let path = public_dir.join(rel_path);
        if path.exists() {
            let digest = file_digest(&path)?;
            let hex = digest.strip_prefix("sha256:").unwrap_or(&digest);
            out.push_str(hex);
            out.push_str("  ");
            out.push_str(rel_path);
            out.push('\n');
        }
    }

    Ok(out)
}

fn publication_subject_digest(parts: &[&str]) -> String {
    let mut hasher = Sha256::new();
    for part in parts {
        hasher.update(part.as_bytes());
        hasher.update(b"\n--jankurai-publication-boundary--\n");
    }
    format!("sha256:{:x}", hasher.finalize())
}

fn digest_text(text: &str) -> String {
    format!("sha256:{:x}", Sha256::digest(text.as_bytes()))
}

fn file_digest(path: &Path) -> Result<String> {
    let bytes = fs::read(path).with_context(|| format!("read {}", path.display()))?;
    Ok(format!("sha256:{:x}", Sha256::digest(bytes)))
}

fn write_text(path: &Path, content: &str) -> Result<()> {
    if let Some(parent) = path.parent() {
        if !parent.as_os_str().is_empty() {
            fs::create_dir_all(parent)
                .with_context(|| format!("create {}", parent.display()))?;
        }
    }
    fs::write(path, content).with_context(|| format!("write {}", path.display()))?;
    Ok(())
}

fn ci_build_url() -> Option<String> {
    let server = std::env::var("GITHUB_SERVER_URL").ok()?;
    let repo = std::env::var("GITHUB_REPOSITORY").ok()?;
    let run = std::env::var("GITHUB_RUN_ID").ok()?;
    Some(format!("{server}/{repo}/actions/runs/{run}"))
}

fn xml_escape(value: &str) -> String {
    value
        .replace('&', "&amp;")
        .replace('<', "&lt;")
        .replace('>', "&gt;")
        .replace('"', "&quot;")
}

fn html_escape(value: &str) -> String {
    xml_escape(value)
}

fn now_string() -> String {
    SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs()
        .to_string()
}
